function MaxSubarraySum_DNC(A, low, high):
    if low == high:
        return A[low]
    mid = floor((low + high) / 2)

    // max sum in left half
    left_sum = MaxSubarraySum_DNC(A, low, mid)
    // max sum in right half
    right_sum = MaxSubarraySum_DNC(A, mid + 1, high)

    // max sum crossing the midpoint
    crossing_sum = MaxCrossingSum(A, low, mid, high)
    return max(left_sum, right_sum, crossing_sum)
function MaxCrossingSum(A, low, mid, high):

    // find max sum of left part (ending at mid)
    left_max = -infinity
    current_sum = 0
    for i from mid down to low:
        current_sum = current_sum + A[i]
        left_max = max(left_max, current_sum)
        
    // find max sum of right part (starting at mid+1)
    right_max = -infinity
    current_sum = 0
    for i from mid + 1 to high:
        current_sum = current_sum + A[i]
        right_max = max(right_max, current_sum)
    return left_max + right_max
