function MaxSubarraySum_DP_Explicit(A):
    n = length(A)
    // initialize DP array
    DP = array of size n
    
    // the max sum ending at index 0 is just A[0]
    DP[0] = A[0]
    
    // initialize overall maximum result
    max_result = DP[0]
    
    // iterate and fill the DP array
    for i from 1 to n-1:
        // start a new subarray at A[i]
        sum_starting_at_i = A[i]
        
        // extend the maximum subarray ending at the previous index
        sum_extending_i = DP[i-1] + A[i]
        
        // DP[i] is the maximum of the two options
        DP[i] = max(sum_starting_at_i, sum_extending_i)
        
        // udate the overall maximum found so far
        max_result = max(max_result, DP[i])
    return max_result
